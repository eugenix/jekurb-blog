### CD pipeline
```yaml

on:
  push:
    branches:
      - master
jobs:
  build:
    name: Deploy
    runs-on: ubuntu-latest
    steps:
      - name: Checkout master
        uses: actions/checkout@v1
        
      - name: Deploy the site
        uses: benmatselby/hugo-deploy-gh-pages@master
        env:
          HUGO_VERSION: 0.74.2

          TOKEN: ${{ secrets.PERSONAL_TOKEN }}
          CNAME: jekurb.dev
```
